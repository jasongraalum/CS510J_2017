001package edu.pdx.cs410J.jgraalum;
002
003import edu.pdx.cs410J.AbstractAirline;
004import edu.pdx.cs410J.AbstractFlight;
005
006import java.util.ArrayList;
007import java.util.Collection;
008
009/**
010 * Airline Class
011 * Created by Jason Graalum on 7/8/17.
012 */
013public class Airline extends AbstractAirline {
014
015    private String airlineName;
016    private ArrayList<Flight> flights = new ArrayList<Flight>();
017
018    /**
019     * Return the airline name
020     * @return      airlineName String
021     */
022    @Override
023    public String getName() {
024
025        return airlineName;
026    }
027
028    /**
029     * Add a flight to the airline
030     * @param flight    AbstractFlight instance
031     */
032    @Override
033    public void addFlight(AbstractFlight flight) {
034        this.flights.add((Flight) flight);
035    }
036
037    /**
038     *
039     * @return      Collection of Flights
040     */
041    @Override
042    public Collection getFlights() {
043
044        return flights;
045    }
046
047    /**
048     *
049     * @return  Created string with airline details.
050     */
051    @Override
052    public String toString() {
053
054        return this.getName()
055                + " with "
056                + this.getFlights().size()
057                + " flights";
058    }
059
060    /**
061     * Create a new Airline instance with the airlineName: name
062     * @param name
063     */
064    Airline(String name) {
065
066        airlineName = name;
067    }
068}
