001package edu.pdx.cs410J.jgraalum;
002
003import edu.pdx.cs410J.AbstractFlight;
004import org.apache.commons.lang3.StringUtils;
005
006/**
007 * <h1>Flight Class</h1>
008 * The Flight Class includes methods and parameters to build flight details for an airline.
009 *
010 * @author Jason Graalum
011 * @version 1.0
012 *
013 */
014public class Flight extends AbstractFlight {
015
016    boolean isValidFlight;
017    Integer number;
018    String sourceAirport;
019    String departureDate;
020    String departureTime;
021    String destinationAirport;
022    String arrivalDate;
023    String arrivalTime;
024
025    /**
026     * Creates a new <code>Flight</code>
027     *
028     * @param flightData
029     *      List of Strings representing flight number, source, departure data and time, destination and arrival date and time.
030     */
031    Flight(String... flightData) {
032        isValidFlight = true;
033        setNumber(flightData[0].trim());
034        setSource(flightData[1].trim());
035        setDepartureString(flightData[2].trim(), flightData[3].trim());
036        setDestination(flightData[4].trim());
037        setArrivalString(flightData[5].trim(), flightData[6].trim());
038    }
039
040    /**
041     * Creates a new empty <code>Flight</code>
042     *
043     */
044    Flight() {
045        this("", "", "", "", "", "", "");
046    }
047
048
049    /**
050     * Returns current flight number
051     * @return Flight number
052     */
053    @Override
054    public int getNumber() {
055        return this.number;
056    }
057
058    /**
059     * Converts flight number string and stores as an integer
060     * @param numberString      String of digits representing the flight number
061     */
062    public void setNumber(String numberString) {
063        if (isValidFlightNumber(numberString))
064            number = Integer.parseInt(numberString);
065        else {
066            System.out.println("Invalid flight number " + numberString + " Number between 1 and 99999 expected.");
067            number = 0;
068            isValidFlight = false;
069        }
070    }
071
072    /**
073     * Returns the source airport string
074     * @return sourceAirport Three letter code of source airport
075     */
076    @Override
077    public String getSource() {
078        return sourceAirport;
079    }
080
081    /**
082     * Sets the source airport code
083     * @param airportCode   Three letter code of source airport
084     */
085    public void setSource(String airportCode) {
086
087        if (isValidAirportCode(airportCode)) {
088            sourceAirport = airportCode;
089        } else {
090            System.out.println("Invalid source airport code: " + airportCode + " A three letter code expected.");
091            sourceAirport = "NAN";
092            isValidFlight = false;
093        }
094    }
095
096    /**
097     * Returns the departure date and time
098     * @return  String of departure data and time in mm/dd/yyyy hh:mm format
099     */
100    @Override
101    public String getDepartureString() {
102        return (departureDate + " " + departureTime);
103    }
104
105    /**
106     * Sets the departure date and time after validating the format
107     * @param date      String in mm/dd/yyyy format
108     * @param time      String in hh:mm formate
109     */
110    public void setDepartureString(String date, String time) {
111        if (isValidDateString(date)) {
112            departureDate = date;
113        } else {
114            System.out.println("Invalid departure date: " + date + " A date of format dd/mm/yyyy expected.");
115            departureDate = "00/00/0000";
116            isValidFlight = false;
117        }
118
119        if (isValidTimeString(time)) {
120            departureTime = time;
121        } else {
122            System.out.println("Invalid departure time: " + time + " A time of format hh:mm expected.");
123            departureTime = "00:00";
124            isValidFlight = false;
125        }
126    }
127
128    /**
129     * Retrieves the destination airport code
130     * @return destinationAirport   Three letter airport code
131     */
132    @Override
133    public String getDestination() {
134        return destinationAirport;
135    }
136
137    /**
138     * Set the destination airport code
139     * @param airportCode   Three letter airport code
140     */
141    public void setDestination(String airportCode) {
142
143        if (isValidAirportCode(airportCode)) {
144            destinationAirport = airportCode;
145        } else {
146            System.out.println("Invalid destination airport code " + airportCode + " A three letter code expected.");
147            destinationAirport = "NAN";
148            isValidFlight = false;
149        }
150    }
151
152    /**
153     * Retrieve arrival date and time
154     * @return  String of arrival data and time in mm/dd/yyyy hh:mm format
155     */
156    @Override
157    public String getArrivalString() {
158        return (arrivalDate + " " + arrivalTime);
159    }
160
161    /**
162     * Sets the arrival date and time after validating the format
163     * @param date      String in mm/dd/yyyy format
164     * @param time      String in hh:mm formate
165     */
166    public void setArrivalString(String date, String time) {
167        if (isValidDateString(date)) {
168            arrivalDate = date;
169        } else {
170            System.out.println("Invalid arrival date " + date + " A date of format dd/mm/yyyy expected.");
171            arrivalDate = "00/00/0000";
172            isValidFlight = false;
173        }
174
175        if (isValidTimeString(time)) {
176            arrivalTime = time;
177        } else {
178            System.out.println("Invalid arrival date " + date + " A time of format hh:mm expected.");
179            arrivalTime = "00:00";
180            isValidFlight = false;
181        }
182    }
183
184    /**
185     * Validates that the flightNumber is a string of numeric digits
186     * @param flightNumber      Flight number string
187     * @return
188     */
189    private boolean isValidFlightNumber(String flightNumber) {
190
191        return (StringUtils.isNumeric(flightNumber));
192    }
193
194    /**
195     * Validate that the airportCode string is three letters.
196     * @param airportCode
197     * @return  Boolean
198     */
199    private boolean isValidAirportCode(String airportCode) {
200        return StringUtils.isAlpha(airportCode) && airportCode.length() == 3;
201    }
202
203    /**
204     * Validate that the string is of mm/dd/yyyy format
205     * @param dateString
206     * @return Boolean
207     */
208    private boolean isValidDateString(String dateString) {
209        String day, month, year;
210        String[] dateFields = dateString.split("/");
211
212        if (dateFields.length == 3) {
213            month = dateFields[0];
214            day = dateFields[1];
215            year = dateFields[2];
216
217            return (StringUtils.isNumeric(day) && StringUtils.isNumeric(month) && StringUtils.isNumeric(year) &&
218                    (Integer.parseInt(day) < 32) &&
219                    (Integer.parseInt(month) < 13) &&
220                    (Integer.parseInt(year) > 1900)
221            );
222        } else {
223            return false;
224        }
225
226    }
227
228    /**
229     * Validate that the string is of hh:mm format
230     * @param timeString
231     * @return Boolean
232     */
233    private boolean isValidTimeString(String timeString) {
234        String hour, minutes;
235        String[] dateFields = timeString.split(":");
236
237        if (dateFields.length == 2) {
238            hour = dateFields[0];
239            minutes = dateFields[1];
240
241            return (StringUtils.isNumeric(hour) && StringUtils.isNumeric(minutes) &&
242                    (Integer.parseInt(hour) < 25) &&
243                    (Integer.parseInt(minutes) < 61)
244            );
245        } else {
246            return false;
247        }
248
249    }
250}
