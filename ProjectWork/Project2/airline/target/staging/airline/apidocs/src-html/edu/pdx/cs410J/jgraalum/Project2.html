001package edu.pdx.cs410J.jgraalum;
002
003import org.apache.commons.lang3.StringUtils;
004import java.util.*;
005
006/**
007 * The main class for the CS410J airline Project
008 */
009public class Project2 {
010
011    private static final ArrayList<String> ARGUMENT_NAMES;
012    private static final ArrayList<String> ARGUMENT_DESCRIPTIONS;
013    private static final ArrayList<String> OPTION_NAMES;
014    private static final ArrayList<String> OPTION_DESCRIPTIONS;
015
016    private static boolean[] optionValues = new boolean[2];
017
018    static {
019        ARGUMENT_NAMES = new ArrayList<String>(Arrays.asList(
020                "name",
021                "flightNumber",
022                "src",
023                "departureTime",
024                "dest",
025                "arriveTime")
026        );
027        ARGUMENT_DESCRIPTIONS = new ArrayList<String>(Arrays.asList(
028                "Then name of the airline",
029                "The flight number",
030                "Three-letter code of departure airport",
031                "Departure data and time (24-hour time)",
032                "Three-letter code of arrival airport",
033                "Arrival date and time (24-hour time)")
034        );
035        OPTION_DESCRIPTIONS = new ArrayList<String>(Arrays.asList(
036                "Prints a description of the new flight",
037                "Prints a README for this project and exits")
038        );
039        OPTION_NAMES = new ArrayList<String>(Arrays.asList(
040                "-print",
041                "-README")
042        );
043    }
044
045
046    public static void main(String[] args) {
047
048        String[] flightData;
049
050        Arrays.fill(optionValues,false);
051        flightData = stripAndParseCmdLindOptions(args);
052
053
054        if(optionValues[1] == true || flightData.length == 0) {
055            printUsageMessageandExit();
056        }
057
058        Airline airline = new Airline(flightData[0]);
059        Flight flight = new Flight(Arrays.copyOfRange(flightData,1, flightData.length));
060
061        airline.addFlight(flight) ;
062
063        if(optionValues[0] == true) {
064            Collection flights = airline.getFlights();
065            Iterator<Flight> flightIter = flights.iterator();
066            while(flightIter.hasNext())
067            {
068                System.out.print(flightIter.next().toString());
069            }
070        }
071
072        System.exit(0);
073  }
074
075  public static void printUsageMessageandExit() {
076
077      Integer NameToDescriptionsPadding = 22;
078
079      String packageName = Project2.class.getCanonicalName();
080
081      System.out.println("usage: java " + packageName + " [options] <args>");
082      System.out.println("  args are (in this order):");
083      Iterator<String> argNamesIter = Project2.ARGUMENT_NAMES.iterator();
084      Iterator<String> argDescriptionsIter = Project2.ARGUMENT_DESCRIPTIONS.iterator();
085      while(argNamesIter.hasNext())
086      {
087          System.out.println(
088                  "    " +
089                  StringUtils.rightPad(argNamesIter.next(),NameToDescriptionsPadding) +
090                  argDescriptionsIter.next()
091          );
092      }
093      System.out.println("  options are (options may appear in any order):");
094      Iterator<String> optionNameIter = Project2.OPTION_NAMES.iterator();
095      Iterator<String> optionDescriptionIter = Project2.OPTION_DESCRIPTIONS.iterator();
096      while(optionNameIter.hasNext())
097      {
098          System.out.println(
099                  "    " +
100                  StringUtils.rightPad( optionNameIter.next(), NameToDescriptionsPadding) +
101                  optionDescriptionIter.next()
102          );
103      }
104      System.out.println("Data and time should be in the format: mm/dd/yyyy hh:mm");
105      System.exit(1);
106  }
107
108  private static boolean isValidOption(String optionName)
109  {
110    return Project2.OPTION_NAMES.contains(optionName);
111  }
112
113
114  private static String[] stripAndParseCmdLindOptions(String[] args)
115  {
116    int optionCount = 0;
117
118    for(String arg: args) {
119        if(arg.charAt(0) == '-')
120        {
121            if(isValidOption(arg)) {
122                optionValues[OPTION_NAMES.indexOf(arg)] = true;
123                optionCount++;
124            }
125            else {
126                IllegalOption(arg);
127            }
128        }
129
130    }
131    return Arrays.copyOfRange(args,optionCount,args.length);
132  }
133
134  private static void IllegalOption(String optionString)
135  {
136    System.out.println("Illegal Option given: " + optionString);
137    System.out.println("");
138    printUsageMessageandExit();
139  }
